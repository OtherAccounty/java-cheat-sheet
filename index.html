LinkedListUtils

import unit4.collectionsLib.Node;

public class LinkedListUtils
{


    public static int size(Node<Integer> head)
    {

        int count = 0; // Initialize count
        Node<Integer> current = head; // Initialize current
        while (current != null  ) {
            count++;
            current = current.getNext();
        }
        return count;

    }



    public static void printList(Node<Integer> head)
    {
        Node<Integer> tnode = head;
        while (tnode != null) {
            System.out.print(tnode.getInfo() + " ");
            tnode = tnode.getNext();
        }
    }


    // Function to delete node at specified position in linked list
    public static Node<Integer> deleteNode(int pos, Node<Integer> orgHead) {


        if (orgHead == null) {
            return null; // Empty list
        }

        Node<Integer> head = copyLinkedList(orgHead);

        if (pos == 0) {
            return head.getNext(); // Delete head
        }

        Node<Integer> prev = null;
        Node<Integer> current = head;
        int index = 0;

        // Traverse the list to find the node at the specified position
        while (current != null && index < pos) {
            prev = current;
            current = current.getNext();
            index++;
        }

        if (current == null) {
            // Position exceeds the size of the list
            throw new IndexOutOfBoundsException("Position is out of bounds");
        }

        // Remove the node at the specified position by adjusting the link of the previous node
        prev.setNext(current.getNext());

        return head;
    }

    public static void deleteNodesWithValue(int value, Node<Integer> orgHead)
    {
        // If the list is empty, return null
        if (orgHead != null) {


            Node<Integer> head = copyLinkedList(orgHead);
            // Track the previous and current nodes
            Node<Integer> prev = null;
            Node<Integer> current = head;

            // Traverse the list
            while (current != null) {
                // If the current node has the value to delete
                if (current.getInfo().equals(value))
                {
                    // If it's the head node, move the head to the next node
                    if (prev == null) {
                        head = current.getNext();
                    } else {
                        // Otherwise, remove the current node by adjusting the links
                        prev.setNext(current.getNext());
                    }
                } else {
                    // Update the previous node if the current node is not deleted
                    prev = current;
                }

                // Move to the next node
                current = current.getNext();
            }

        }
    }

    public static Node<Integer> copyLinkedList(Node<Integer> org)
    {
        if (org == null) {
            return null;
        }

        // Copy the current node
        Node<Integer> newNode = new Node<>(org.getInfo());

        // Recursively copy the remaining list and set it as the next of the new node
        newNode.setNext(copyLinkedList(org.getNext()));

        return newNode;
    }


    // Function to add a node at a specific index in the linked list
    public static Node<Integer> addAtIndexInNewList(int index, int value, Node<Integer> orgHead) {

        Node<Integer> head = copyLinkedList(orgHead);

        if (index == 0) {
            return new Node<>(value, head);
        }

        Node<Integer> prev = null;
        Node<Integer> current = head;

        int currentIndex = 0;

        while (current != null && currentIndex < index) {
            prev = current;
            current = current.getNext();
            currentIndex++;
        }

        if (currentIndex != index) {
            throw new IndexOutOfBoundsException("Index is out of bounds");
        }

        // Insert the new node after the found node
        prev.setNext(new Node<>(value, current));

        return head;
    }

    // reminder that the index starts from 0
    // RETURN -1 IF NOT FOUND
    public static int findIndexOfNumberFromListStart(Node<Integer> lst, int num)
    {
        int index =0;
        boolean found = false;
        Node<Integer> curr = lst;

        while(curr != null && !found)
        {
            if(curr.getInfo() == num)
            {
                found = true;
            }
            else
            {
                index++;
                curr = curr.getNext();
            }
        }

        if(!found)
        {
            index = -1;
        }
        return index;

    }

    // function finds the index of the number form the back of the list
    // list: 7->2->8->20->18->19->20->20->5->-3->0->9
    // INDEX FROM END 8 = 10
    public static int findIndexOfNumberFromListEnd(Node<Integer> lst, int num)
    {
        int index =0;
        int foundIndex = -1;
        boolean found = false;
        Node<Integer> curr = lst;

        while(curr != null)
        {
            if(curr.getInfo() == num)
            {
                foundIndex = index;
                found = true;
            }
            index++;

            curr = curr.getNext();

        }

        if(found==true)
        {
             foundIndex = index - foundIndex;
        }

        return foundIndex;

    }

}

Queue

Simple Queue is a linear data structure that follows the First-In-First-Out (FIFO) principle, where elements are added to the rear (back) and removed from the front (head). 


import java.util.Random;

public class QueueUtil {

    // helper method to copy the queue
  public static Queue<Integer> copyQueue(Queue<Integer> original)
 {
        Queue<Integer> copy = new Queue<>();
        Queue<Integer> tempQueue = new Queue<>();

        // Copy the elements while keeping the original intact
        while (!original.isEmpty())
			  {
            int element = original.remove();
            copy.insert(element);
            tempQueue.insert(element);
        }

        // Restore the original queue
        while (!tempQueue.isEmpty())
        {
            original.insert(tempQueue.remove());
        }

        return copy;
    }


    // function builds a queue made by 10 random numbers from 10 to -10 and returns it
    public static Queue<Integer> queue10_10Random ()
    {
        final int MAX_RAND = 10;
        final int MIN_RAND = -10;

       Random rand = new Random();
       Queue<Integer> randomQueue = new Queue<>();

       int currRand = 0;
       for(int i=0; i<10; i++)
       {
           currRand = rand.nextInt(MAX_RAND - MIN_RAND + 1) + MIN_RAND;

           randomQueue.insert(currRand);
       }

       return randomQueue;
    }

    // returns amount of members in the queue
    public static  int queueSize (Queue<Integer> originalQ )
    {
        Queue<Integer> q = copyQueue(originalQ);
        int count=0;

        while(!q.isEmpty())
        {
            q.remove();
            count++;

        }

        return count;

    }


    // function receives a queue<integer> and returs  the sum of its members
    public static  int queueSum(Queue<Integer> originalQ )
    {
        Queue<Integer> q = copyQueue(originalQ);

        int sum=0;
        while(!q.isEmpty())
        {
            sum+=q.remove();
        }
        return sum;
    }

    // function receies a queue of integers and returns the sum of the even digits in the queue
    public static  int queueSumEven (Queue<Integer> originalQ )
    {
        Queue<Integer> q = copyQueue(originalQ);

        int sum=0;
        int curr=0;

        while(!q.isEmpty())
        {

            curr = q.remove();

            if (curr % 2 == 0)
            {
                sum+=curr;
            }
        }

        return sum;
    }

    // function returns the sum of the first and last digits in the given queue
    public static int queueSumFirstLast(Queue<Integer> originalQ )
    {
        Queue<Integer> q = copyQueue(originalQ);

        int sum=0;
        int curr=0;

        sum += q.remove();

        while(!q.isEmpty())
        {
            curr = q.remove();
        }
        sum+=curr;
        return sum;
    }

    // receives a queue and returns the highest number on it
    public static  int queueMax( Queue<Integer> originalQ )
    {
        Queue<Integer> q = copyQueue(originalQ);

        int max=Integer.MIN_VALUE;
        int curr=0;

        while(!q.isEmpty())
        {

            curr = q.remove();

            if (curr>max)
            {
                max=curr;
            }
        }

        return max;
    }



    // Function to delete element at specified index from Queue and return a new Queue
    public static Queue<Integer> deleteAtIndex(int index, Queue<Integer> orgQ) {

        Queue<Integer> copyOrg = copyQueue(orgQ);
        Queue<Integer> delQ = new Queue<>();

        int queueSize = queueSize(orgQ); // minus one as it will take out an index
        for(int i=0; i<queueSize; i++)
        {
            if(i == index) // if its in the index, jump it
            {
                copyOrg.remove();
            }
            else
            {
                delQ.insert(copyOrg.remove());
            }

        }


        return delQ;
    }

    public static Queue<Integer> insertAtIndex(int index, int element, Queue<Integer> orgQ) {
        Queue<Integer> copyOrg = copyQueue(orgQ);
        Queue<Integer> insQ = new Queue<>();

        int queueSize = queueSize(orgQ);

        for (int i = 0; i <= queueSize; i++) {
            if (i == index) {
                insQ.insert(element);
            }
            if (!copyOrg.isEmpty()) {
                insQ.insert(copyOrg.remove());
            }
        }

        return insQ;
    }

}
